--- layout: default --- {%- comment -%} Order comes from _data/photo_order.yml (newest first by EXIF capture date, via bin/generate-photo-order.py). Any file on disk that isn't listed is appended, so a stale order file never hides a photograph. {%- endcomment -%} {%- assign photo_files = site.static_files | where_exp: "f", "f.path contains '/assets/images/photography/'" -%} {%- assign order = site.data.photo_order -%} {%- assign ordered_names = "" | split: "" -%} {%- for name in order -%} {%- assign want = "/assets/images/photography/" | append: name -%} {%- for f in photo_files -%} {%- if f.path == want -%} {%- assign ordered_names = ordered_names | push: name -%} {%- endif -%} {%- endfor -%} {%- endfor -%} {%- for f in photo_files -%} {%- assign fname = f.path | split: '/' | last -%} {%- unless ordered_names contains fname -%} {%- assign ordered_names = ordered_names | push: fname -%} {%- endunless -%} {%- endfor -%} {% include viewer.html base="/assets/images/photography/" %}
